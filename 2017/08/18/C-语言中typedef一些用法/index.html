typedef一些用法

重命令

重命名之后，程序看起来更简洁

typedef unsigned char uchar;

typedef struct
{
    int t1;
    double e1;
}di_st，*p_di_st;

使用的时候直接这样用。

p_di_st p = NULL;
p = malloc(sizeof(di_st));

跨平台

用typedef来定义与平台无关的类型。

#if defined(SOCKET) || defined(_WINDOWS)
typedef SOCKET    XXX_SOCKET;
#else
typedef int    XXX_SOCKET;
#endif

例如Windows使用的套接字类型和Linux不一样。

用做回掉函数

/*************************************************************************
    > File Name: call.c
    > Author: Lin
    > Mail: 不想写邮箱
    > Created Time: Thu 15 Jun 2017 10:25:11 AM CST
 ************************************************************************/

#include <stdio.h>
#include <unistd.h>
#include <string.h>
#include <stdlib.h>

typedef void (*call_back_t)(void *);

void test(void *str)
{
    printf("%s\n",(char *)str);
}

void hhh(call_back_t c,void *p)
{
    c(p);
}

int main(int argc, const char *argv[])
{
    char *a = "1234678";
    hhh(&test,(void *)a);    
    return 0;
}

# gcc call.c -o call
# ./call
12345678
