1. #if、#elif、#else和#endif

方式1

#if 常量表达式
    程序1
#endif

方式2

#if 常量表达式1
    程序1
#elif defined(常量表达式2)
    程序2
。。。
#elif defined(常量表达式n)
    程序n
#else
    程序
#endif

功能：当常量表达式n为非0（”逻辑真时”）时，编译程序n。

常量表达式：可以是逻辑运算符”!”、”==”、”!=”和”||”

#include <stdio.h>

#define TEST002
static void test001()
{
    printf("test001\n");
}

static void test002()
{
    printf("test002\n");
}

static void test003()
{
    printf("test003\n");
}

int main(int argc, const char *argv[])
{
#if TEST001
    test001();
#elif defined(TEST002)
    test002();
#endif

#if !defined(TT)
    test003();
#endif
    return 0;
}

2. #ifdef #else #endif

#ifdef 宏
    程序1
#else
    程序2
#endif

如果宏被定义，运行程序1，否则运行程序2

eg:

#include <stdio.h>
//#define __DEBUG
int main(int argc, const char *argv[])
{
#ifdef __DEBUG
    printf("TTTT\n");
#else
    printf("SSSS\n");
#endif
    return 0;
}

3. #ifndef #else #endif

#ifndef 宏
    程序1
#else
    程序2
#endif

如果宏没有被定义，则运行程序1，否则运行程序2

#include <stdio.h>
//#define __DEBUG
int main(int argc, const char *argv[])
{
#ifndef __DEBUG
        printf("TTTT\n");
#else
        printf("SSSS\n");
#endif
    return 0;
}
